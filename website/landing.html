import { TONLDocument } from 'tonl';

// Load or create document
const doc = await TONLDocument.fromFile('data.tonl');

// Query with filters
const admins = doc.query(
  'users[?(@.role == "admin")]'
);

// Modify data
doc
  .set('users[0].age', 31)
  .set('users[0].verified', true);

// Create indices for fast lookups
doc.createIndex({
  name: 'byId',
  fields: 'id',
  unique: true
});

// Find with O(1) performance
const user = doc
  .getIndex('byId')
  .find(123);

// Save atomically
await doc.save('data.tonl');
